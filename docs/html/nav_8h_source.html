00144    {
00145       // Only update if we are the master node
00146       if (!mNavData.isLocal())
00147       {
00148          return;
00149       }
00150       
00151       // Clamp delta
00152       float delta = 1.0f;
00153 
00154       // ---------- ROTATION ----------- //
00155       // Scale the rotation velocity (slerp) to get a time based change
00156       gmtl::Quatf qrot = gmtl::make<gmtl::Quatf>(mRotVelocity);
00157       gmtl::Quatf scaled_qrot, src_rot;   // scaled rotation and Identity rotation
00158       gmtl::Matrix44f delta_rot;          // The delta rotation to use
00159 
00160       // Only compute if we don't have identity rotation
00161       if (!gmtl::isEqual(gmtl::MAT_IDENTITY44F, mRotVelocity, 0.001f))
00162       {
00163          gmtl::slerp(scaled_qrot, delta, src_rot, qrot);
00164          gmtl::set(delta_rot, scaled_qrot);
00165       }
00166 
00167       if (mMode == WALK)
00168       {
00169          gmtl::EulerAngleXYZf euler( 0.0f, gmtl::makeYRot(delta_rot), 0.0f );// Only allow Yaw (rot y)
00170          delta_rot = gmtl::makeRot<gmtl::Matrix44f>( euler );
00171       }
00172       gmtl::postMult(mNavData->mCurPos, delta_rot);
00173 
00174       // ------- TRANSLATION ---- //
00175       gmtl::Vec3f trans_delta;
00176       if (mMode == WALK)
00177       {
00178          mVelocity[1] = 0.0f;
00179       }
00180       trans_delta =  mVelocity * delta;
00181 
00182       // Post multiply the delta translation
00183       gmtl::Matrix44f trans_matrix = gmtl::makeTrans<gmtl::Matrix44f>(trans_delta);
00184       gmtl::postMult(mNavData->mCurPos, trans_matrix);
00185       
00186       // ------- TILT ROTATION ---- //
00187        // Scale the rotation velocity (slerp) to get a time based change
00188        gmtl::Quatf qrotTilt = gmtl::make<gmtl::Quatf>(mTiltVelocity);
00189        gmtl::Quatf scaled_qrotTilt, src_rotTilt;   // scaled rotation and Identity rotation
00190        gmtl::Matrix44f delta_rotTilt;          // The delta rotation to use
00191        
00192        // Only compute if we don't have identity rotation
00193        if (!gmtl::isEqual(gmtl::MAT_IDENTITY44F, mTiltVelocity, 0.001f))
00194        {
00195            gmtl::slerp(scaled_qrotTilt, delta, src_rotTilt, qrotTilt);
00196            gmtl::set(delta_rotTilt, scaled_qrotTilt);
00197        }
00198        gmtl::postMult(mTiltData->mCurPos, delta_rotTilt);
00199 
00200       // Reset velocity vectors
00201       mVelocity = gmtl::Vec3f(0.0, 0.0, 0.0);
00202       mRotVelocity = gmtl::MAT_IDENTITY44F;
